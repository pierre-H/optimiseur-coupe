84 int MoteurCalculs::moteurCombinaisons (Combinaison& l)
85 {

97  p->setPosition(i); // création d'1 nvelle paire

101  case 0 :
102  cpt++; // nouvelle possibilité créé rajoutée à la liste

104  break;
105  case 1 : // tailler la branche qui dépasse la longueur de la barre
106  break;
107  default : // pas rajouter mais continuer de mouliner
108  cpt += moteurCombinaisons(*lp);
109  break;
110  }
111  }
112  return cpt;
